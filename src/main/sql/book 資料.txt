CREATE TABLE book
(
    book_id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    book_name VARCHAR(256) NOT NULL UNIQUE KEY,
    author VARCHAR(256) NOT NULL,
    price INT NOT NULL,
    stock INT NOT NULL
);
